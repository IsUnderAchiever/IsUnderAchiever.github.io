    private void loadBean(File file) throws Exception {
        // 1.åˆ¤æ–­å½“å‰æ˜¯å¦æ˜¯æ–‡ä»¶å¤¹
        if (file.isDirectory()) {
            // 2.è·å–æ–‡ä»¶å¤¹é‡Œçš„æ‰€æœ‰å†…å®¹(æ–‡ä»¶å¤¹ã€æ–‡ä»¶)
            File[] childrenFiles = file.listFiles();
            // 3.è‹¥æ–‡ä»¶å¤¹é‡Œä¸ºç©ºï¼Œç›´æ¥è¿”å›
            if (childrenFiles == null || childrenFiles.length == 0) {

            // 4.æ–‡ä»¶å¤¹ä¸ä¸ºç©ºï¼Œéå†æ–‡ä»¶å¤¹æ‰€æœ‰å†…å®¹
            for (File child : childrenFiles) {
                // 4.1éå†å¾—åˆ°æ¯ä¸ªFileå¯¹è±¡ï¼Œé€’å½’åˆ¤æ–­æ–‡ä»¶å¤¹é‡Œæ˜¯å¦è¿˜æœ‰æ–‡ä»¶å¤¹...
                if (child.isDirectory()) {
                    // é€’å½’
                    loadBean(child);
                } else {
                    // 4.2éå†å¾—åˆ°Fileå¯¹è±¡ä¸æ˜¯æ–‡ä»¶å¤¹ï¼Œæ˜¯æ–‡ä»¶
                    // 4.3å¾—åˆ°ã€åŒ…è·¯å¾„+ç±»åç§°ã€‘éƒ¨åˆ†-å­—ç¬¦ä¸²æˆªå–
                    String pathWithClass = child.getAbsolutePath().substring(rootPath.length() - 1);
                    // 4.4åˆ¤æ–­å½“å‰ç±»å‹æ˜¯å¦æ˜¯.class
                    if (pathWithClass.contains(".class")) {
                        // 4.5å¦‚æœæ˜¯.classç±»å‹ï¼Œå°±æŠŠ\æ›¿æ¢æˆ. å¹¶ å»æ‰åç¼€.class
                        // com.example.service.UserServiceImpl  å› ä¸ºåªæœ‰è·å–ç±»å…¨è·¯å¾„åæ‰èƒ½ä½¿ç”¨åå°„
                        String allName = pathWithClass.replaceAll("\\\\", ".").replace(".class", "");
                        // 4.6åˆ¤æ–­ç±»ä¸Šæ˜¯å¦æœ‰æ³¨è§£@Beanï¼Œå¦‚æœæœ‰ï¼Œå®ä¾‹åŒ–è¿‡ç¨‹
                        // 4.6.1è·å–ç±»çš„classå¯¹è±¡
                        Class<?> clazz = Class.forName(allName);
                        // 4.6.2åˆ¤æ–­ï¼Œå¦‚æœä¸æ˜¯æ¥å£ï¼Œåˆ™å®ä¾‹åŒ–
                        if (!clazz.isInterface()) {
                            // 4.6.3åˆ¤æ–­ï¼Œç±»ä¸Šæ˜¯å¦æœ‰æ³¨è§£@Bean
                            Bean annotation = clazz.getAnnotation(Bean.class);
                            if (annotation != null) {
                                // ç±»ä¸Š@Beanæœ‰æ³¨è§£
                                // 4.6.4å®ä¾‹åŒ–
                                Object instance = clazz.getConstructor().newInstance();
                                // 4.7æŠŠå¯¹è±¡å®ä¾‹åŒ–ä¹‹åï¼Œæ”¾åˆ°mapé›†åˆbeanFactory
                                // 4.7.1å°ç»†èŠ‚:å¦‚æœæœ‰æ¥å£ï¼Œå°±æ”¾æ¥å£classä½œä¸ºkeyï¼Œå¦‚æœæ²¡æœ‰ï¼Œå°±è®©è‡ªå·±classä½œä¸ºkey
                                if (clazz.getInterfaces().length > 0) {
                                    beanFactory.put(clazz.getInterfaces()[0], instance);
                                } else {
                                    beanFactory.put(clazz, instance);
                                }
                            }
                        }
                    }
                }
            }
        }
    }
